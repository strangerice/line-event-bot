⑤ 完全コード（コピペ可）
server.js
const express = require("express")
const line = require("@line/bot-sdk")
const cron = require("node-cron")
const axios = require("axios")

const app = express()

const config = {
 channelAccessToken: process.env.LINE_TOKEN,
 channelSecret: process.env.LINE_SECRET
}

const client = new line.Client(config)

app.post("/webhook", line.middleware(config), async (req, res) => {

 const events = req.body.events

 for(const event of events){

  if(event.type !== "message") continue
  if(event.message.type !== "text") continue

  const text = event.message.text

  const match = text.match(/イベント「(.+?)」.*日時[:：]\s*(\d{4}\/\d{2}\/\d{2}\s*\d{2}:\d{2})/)

  if(match){

   const title = match[1]
   const date = new Date(match[2])

   await saveEvent(title,date,event.source.groupId)

   await client.replyMessage(event.replyToken,{
     type:"text",
     text:`イベント登録しました\n${title}\n${date}`
   })

  }

 }

 res.sendStatus(200)

})

async function saveEvent(title,date,groupId){

 const data={
  title,
  date:date.toISOString(),
  groupId
 }

 await axios.post(process.env.SHEET_API,data)

}

async function sendReminder(event){

 const text = `🔔イベントリマインド\n${event.title}\n日時:${event.date}`

 await client.pushMessage(event.groupId,{
   type:"text",
   text
 })

}

cron.schedule("0 0 * * *", async ()=>{

 const events = await getEvents()

 const now = new Date()

 for(const event of events){

   const eventDate = new Date(event.date)

   const diff = (eventDate-now)/1000/60/60/24

   if(Math.floor(diff)==14){
     await sendReminder(event)
   }

   if(Math.floor(diff)==7){
     await sendReminder(event)
   }

 }

})

cron.schedule("0 9 * * *", async ()=>{

 const events = await getEvents()
 const today = new Date().toISOString().slice(0,10)

 for(const event of events){

   if(event.date.startsWith(today)){
     await sendReminder(event)
   }

 }

})

app.listen(3000,()=>console.log("bot running"))
